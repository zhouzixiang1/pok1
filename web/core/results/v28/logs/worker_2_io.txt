def card_rank(card):
    return card // 4 + 2

def card_suit(card):

def is_suited(c1, c2):

def rank_gap(c1, c2):

def preflop_hand_strength(hand):
    """Evaluate 2-card preflop hand strength on a 0–1 scale.
    Uses a simplified Chen-like formula."""
    c1, c2 = hand
    r1, r2 = card_rank(c1), card_rank(c2)
    s1, s2 = card_suit(c1), card_suit(c2)

    high = max(r1, r2)
    low  = min(r1, r2)
    suited = (s1 == s2)
    gap = high - low
    is_pair = (r1 == r2)

    if is_pair:
        score = 0.50 + (high - 2) / 12.0 * 0.50
    else:
        score = (high - 2) / 12.0 * 0.45 + (low - 2) / 12.0 * 0.20 + 0.15
        if suited:
            score += 0.06
        if gap <= 1:
            score += 0.05
        elif gap == 2:
            score += 0.03
        elif gap == 3:
            score += 0.01
        if gap >= 5:
            score -= 0.04

    return min(1.0, max(0.0, score))

def _evaluate_five(cards):
    """Evaluate a 5-card hand. Returns (rank, tiebreaker)."""
    ranks = sorted([card_rank(c) for c in cards], reverse=True)
    suits = [card_suit(c) for c in cards]

    is_flush = len(set(suits)) == 1

    unique_ranks = sorted(set(ranks), reverse=True)
    is_straight = False
    straight_high = 0

    if len(unique_ranks) == 5:
        if unique_ranks[0] - unique_ranks[4] == 4:
            is_straight = True
            straight_high = unique_ranks[0]
        if unique_ranks == [14, 5, 4, 3, 2]:
            is_straight = True
            straight_high = 5

    freq = Counter(ranks)
    freq_vals = sorted(freq.values(), reverse=True)
    freq_items = sorted(freq.items(), key=lambda x: (x[1], x[0]), reverse=True)

    if is_straight and is_flush:
        return (8, [straight_high])
    if freq_vals == [4, 1]:
        return (7, [freq_items[0][0], freq_items[1][0]])
    if freq_vals == [3, 2]:
        return (6, [freq_items[0][0], freq_items[1][0]])
    if is_flush:
        return (5, ranks)
    if is_straight:
        return (4, [straight_high])
    if freq_vals == [3, 1, 1]:
        kicker = sorted([r for r, f in freq.items() if f == 1], reverse=True)
        return (3, [freq_items[0][0]] + kicker)
    if freq_vals == [2, 2, 1]:

        return (2, pairs + kicker)

        kicker = sorted([r for r, f in freq.items() if f == 1], reverse=True)
        return (1, [freq_items[0][0]] + kicker)
    return (0, ranks)


def evaluate_hand(hole_cards, community_cards):
    """Evaluate best 5-card hand. Returns (rank, tiebreaker)."""
    all_cards = list(hole_cards) + list(community_cards)
    if len(all_cards) < 5:
        return _partial_hand_rank(hole_cards, community_cards)

    best_rank = -1
    best_tb = []
    for combo in combinations(all_cards, 5):
        rank, tb = _evaluate_five(list(combo))
        if (rank, tb) > (best_rank, best_tb):
            best_rank = rank
            best_tb = tb
    return (best_rank, best_tb)


def _partial_hand_rank(hole_cards, community_cards):
    """Simplified hand rank for incomplete boards."""
    if len(community_cards) == 0:
        strength = preflop_hand_strength(hole_cards)
        return (0, [int(strength * 14)])

    all_cards = list(hole_cards) + list(community_cards)
    ranks = [card_rank(c) for c in all_cards]
    freq = Counter(ranks)

    best_pair = 0
    for r, count in freq.items():
        if count >= 3:
            return (3, [r])
        if count >= 2 and r > best_pair:
            best_pair = r
    if best_pair > 0:
        return (1, [best_pair])
    return (0, sorted(ranks, reverse=True))


def hand_strength_normalized(hole_cards, community_cards):
    """Return normalized hand strength 0–1."""
    rank, tiebreaker = evaluate_hand(hole_cards, community_cards)
    base = rank / 8.0
    tb = 0
    if tiebreaker:
        tb = max(0, min(1, (tiebreaker[0] - 2) / 12.0))
    return min(1.0, base * 0.7 + tb * 0.3)

def detect_draws(hole_cards, community_cards):
    """Detect flush and straight draws. Returns draw strength 0–0.5."""
    if len(community_cards) < 3:
        return 0.0

    all_cards = list(hole_cards) + list(community_cards)
    draw_strength = 0.0

    suits = [card_suit(c) for c in all_cards]
    suit_freq = Counter(suits)
    max_suit = max(suit_freq.values())

    if max_suit == 4:
        draw_strength += 0.20
    elif max_suit == 3 and len(community_cards) <= 4:
        draw_strength += 0.05

    ranks = sorted(set([card_rank(c) for c in all_cards]))
    if len(ranks) >= 4:
        for i in range(len(ranks) - 3):
            window = ranks[i:i + 4]
            span = window[-1] - window[0]
            if span <= 4:
                if span == 3:
                    draw_strength += 0.18
                else:
                    draw_strength += 0.08

    if 14 in ranks and any(r in ranks for r in [2, 3, 4, 5]):
        needed = sum(1 for r in [2, 3, 4, 5] if r not in ranks)
        if needed <= 1:
            draw_strength += 0.12

    return min(0.50, draw_strength)

def parse_game_state(data):
    """Parse input JSON into a structured game state dict."""
    state = {}

    state['hand'] = data.get('hand', data.get('hole_cards', []))
    state['community'] = data.get('public_card', data.get('community_cards', data.get('board', [])))
    if state['community'] is None:
        state['community'] = []

    history = data.get('history', {})
    state['history'] = history

    if isinstance(history, dict):
        state['position'] = history.get('mypos', data.get('my_pos', 0))
        state['log'] = history.get('log', [])
    else:
        state['position'] = data.get('my_pos', 0)
        state['log'] = []

    state['pot'] = data.get('pot', 0)
    state['my_bet'] = data.get('my_bet', 0)
    state['opp_bet'] = data.get('opp_bet', 0)

    bets = data.get('bet', [0, 0])
    if isinstance(bets, list) and len(bets) >= 2:
        pos = state['position']
        state['my_bet'] = bets[pos] if pos < len(bets) else 0
        state['opp_bet'] = bets[1 - pos] if (1 - pos) < len(bets) else 0

    state['my_chips'] = data.get('my_chips', data.get('my_stack', STARTING_CHIPS))
    state['opp_chips'] = data.get('opp_chips', data.get('opp_stack', STARTING_CHIPS))

    if not state['community']:
        state['street'] = 'preflop'
    elif len(state['community']) == 3:
        state['street'] = 'flop'
    elif len(state['community']) == 4:
        state['street'] = 'turn'
    else:
        state['street'] = 'river'

    state['call_amount'] = max(0, state['opp_bet'] - state['my_bet'])
    state['pot_size'] = state['pot']

    return state

    effective = min(1.0, strength * pos_mult)

    if street == 'preflop':
        return _decide_preflop(state, effective)
    return _decide_postflop(state, effective, strength)


def _decide_preflop(state, strength):
    """Preflop decision logic."""
    position = state['position']
    call_amount = state['call_amount']
    pot_size = state['pot_size']
    my_chips = state['my_chips']
    facing_raise = call_amount > 0

    if position == 0:  # SB — acts first preflop
        if not facing_raise:
            if strength >= PREFLOP_RAISE_THRESHOLD_SB:
                raise_amt = int(max(MIN_RAISE_MULTIPLE * BLIND_BIG, pot_size * RAISE_MULTIPLIER))
                raise_amt = min(raise_amt, my_chips)
                if strength >= ALL_IN_THRESHOLD:
                    return -2
                return raise_amt
            if strength >= PREFLOP_CALL_THRESHOLD_SB:
                return 0
            if random.random() < BLUFF_FREQUENCY * 0.3:
                raise_amt = int(pot_size * RAISE_MULTIPLIER)
                return min(raise_amt, my_chips)
            return 0
        else:
            if strength >= THREE_BET_MIN_STRENGTH:
                raise_amt = int(call_amount * THREE_BET_SIZE_BB)
                return min(raise_amt, my_chips)
            if strength >= PREFLOP_CALL_THRESHOLD_SB:
                return 0
            if strength >= LIGHT_3BET_STRENGTH and random.random() < 0.30:
                raise_amt = int(call_amount * THREE_BET_SIZE_BB)
                return min(raise_amt, my_chips)
            return -1
    else:  # BB
        if not facing_raise:
            if strength >= PREFLOP_RAISE_THRESHOLD_BB:
                raise_amt = int(max(MIN_RAISE_MULTIPLE * BLIND_BIG, pot_size * RAISE_MULTIPLIER))
                raise_amt = min(raise_amt, my_chips)
                if strength >= ALL_IN_THRESHOLD:
                    return -2
                return raise_amt
            return 0
        else:
            if strength >= FOUR_BET_MIN_STRENGTH:
                raise_amt = int(call_amount * 2.5)
                return min(raise_amt, my_chips)
            if strength >= THREE_BET_MIN_STRENGTH:
                raise_amt = int(call_amount * THREE_BET_SIZE_BB)
                return min(raise_amt, my_chips)
            if strength >= PREFLOP_CALL_THRESHOLD_BB:
                return 0
            return -1


def _decide_postflop(state, effective, raw_strength):
    """Postflop decision logic."""
    street = state['street']
    position = state['position']
    call_amount = state['call_amount']
    pot_size = state['pot_size']
    my_chips = state['my_chips']
    facing_bet = call_amount > 0

    if call_amount > 0 and pot_size > 0:
        pot_odds = call_amount / (pot_size + call_amount)
    else:
        pot_odds = 0

    if call_amount > 0:
        ev_call = effective * (pot_size + call_amount) - (1 - effective) * call_amount
    else:
        ev_call = 0

    # --- Monster hands (slow play / trap) ---
    if raw_strength >= SLOW_PLAY_THRESHOLD:
        if facing_bet:
            if random.random() < TRAP_FREQUENCY:
                return 0
            raise_amt = int(pot_size * VALUE_BET_POT_FRACTION * 1.5)
            return min(raise_amt, my_chips)
        if random.random() < TRAP_FREQUENCY:
            return 0
        bet = int(pot_size * THIN_VALUE_BET_FRACTION)
        return min(max(bet, int(MIN_RAISE_MULTIPLE * BLIND_BIG)), my_chips)

    # --- Strong hands (value bet) ---
    if raw_strength >= STRONG_HAND_THRESHOLD:
        if facing_bet:
            if ev_call > 0:
                raise_amt = int(pot_size * VALUE_BET_POT_FRACTION)
                return min(raise_amt, my_chips)
            return 0
        if raw_strength >= OVERBET_THRESHOLD:
            bet = int(pot_size * OVERBET_FRACTION)
        else:
            bet = int(pot_size * VALUE_BET_POT_FRACTION)
        return min(bet, my_chips)

    # --- Medium hands ---
    if raw_strength >= MEDIUM_HAND_THRESHOLD:
        if facing_bet:
            if pot_odds < POT_ODDS_CALL_THRESHOLD or ev_call > 0:
                return 0
            if call_amount > my_chips * COMMIT_THRESHOLD:
                return -1
            return 0
        cbet_freq = CBET_FREQUENCY if position == 1 else CBET_FREQUENCY_OOP
        if random.random() < cbet_freq:
            bet = int(pot_size * BLUFF_BET_POT_FRACTION)
            return min(max(bet, int(MIN_RAISE_MULTIPLE * BLIND_BIG)), my_chips)
        return 0

    # --- Weak-medium hands ---
    if raw_strength >= WEAK_HAND_THRESHOLD:
        if facing_bet:
            if ev_call > 0 and pot_odds < POT_ODDS_CALL_THRESHOLD:
                return 0
            if call_amount < pot_size * 0.20:
                return 0
            if random.random() < FOLD_TO_RAISE_THRESHOLD:
                return -1
            return 0
        bluff_freq = BLUFF_FREQUENCY if position == 1 else BLUFF_FREQUENCY * 0.6
        if random.random() < bluff_freq:
            bet = int(pot_size * BLUFF_BET_POT_FRACTION)
            return min(bet, my_chips)
        return 0

    # --- Weak hands ---
    if facing_bet:
        if ev_call > POT_ODDS_CALL_THRESHOLD:
            return 0
        return -1
    bluff_freq = BLUFF_FREQUENCY * 0.7
    if random.random() < bluff_freq:
        bet = int(pot_size * BLUFF_BET_POT_FRACTION)
        return min(bet, my_chips)
    return 0


# ============================================================
# MAIN ENTRY POINT
# ============================================================

def main():
    try:
        raw = sys.stdin.read()
        data = json.loads(raw)
    except Exception:
        print(json.dumps({"response": 0}))
        return

    state = parse_game_state(data)

    try:
        action = decide(state)
    except Exception:
        action = 0

    if action is None:
        action = 0
    if not isinstance(action, int):
        action = int(action)

    my_chips = state.get('my_chips', STARTING_CHIPS)
    if action > my_chips:
        action = my_chips

    print(json.dumps({"response": action}))


if __name__ == "__main__":
    main()
