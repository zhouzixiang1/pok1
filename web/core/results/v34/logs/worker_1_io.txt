    if round_idx < 2:

        return False
    
    # Never fold decent pairs
    if pair_profile is not None and pair_profile["made_class"] == 1:
        if pair_profile["pair_type"] in ("overpair", "top_pair", "pocket_pair"):
            return False
    
    # Never fold with draws
    if draw_strength >= 0.10:
        return False
    
    size_bucket = bet_size_bucket(spot_info["last_raise_pot_ratio"])
    
    # TURN folds (round_idx == 2)
    if round_idx == 2:
        # Fold pure air (no pair, no draw) to medium+ bets
        if made_strength < 0.18 and draw_strength < 0.08:
            if size_bucket in ("medium", "large"):
                return True
        # Fold weak pairs (bottom/under/board) to large bets on dynamic boards
        if pair_profile is not None and pair_profile["pair_type"] in ("bottom_pair", "underpair", "board_pair"):
            if made_strength < 0.30 and board_texture is not None and board_texture["dynamic"]:
                if size_bucket == "large":
                    return True
    
    # RIVER folds (round_idx == 3)
    if round_idx == 3:
        # Fold pure air to medium+ bets
        if made_strength < 0.18 and draw_strength < 0.08:
            if size_bucket in ("medium", "large"):
                return True
        # Fold weak pairs to large bets
        if pair_profile is not None and pair_profile["pair_type"] in ("middle_pair", "bottom_pair", "underpair", "board_pair"):
            if made_strength < 0.30:
                if size_bucket == "large":
                    if nutted_risk.get("risk", 0) < 0.04:
                        return True
        # Fold medium-strength no-showdown hands to large bets
        if 0.18 <= made_strength < 0.35 and draw_strength < 0.08:
            if size_bucket == "large" and not (pair_profile and pair_profile["pair_type"] == "top_pair"):
                return True
    
    return False
